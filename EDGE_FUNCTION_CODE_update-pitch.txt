import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
  'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
}

serve(async (req) => {
  // Handle CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders })
  }

  try {
    // Get token from Authorization header
    const authHeader = req.headers.get('Authorization')
    if (!authHeader) {
      return new Response(
        JSON.stringify({ error: 'Missing authorization token' }),
        { status: 401, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }

    const token = authHeader.replace('Bearer ', '').trim()

    // Initialize Supabase
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!

    // Verify token using REST API directly (avoids JWT validation)
    const verifyResponse = await fetch(
      `${supabaseUrl}/rest/v1/user_profiles?access_token=eq.${encodeURIComponent(token)}&role=eq.owner&select=id,role`,
      {
        headers: {
          'apikey': supabaseServiceKey,
          'Authorization': `Bearer ${supabaseServiceKey}`,
          'Content-Type': 'application/json',
        }
      }
    )
    
    const verifyData = await verifyResponse.json()
    const profile = Array.isArray(verifyData) && verifyData.length > 0 ? verifyData[0] : null

    if (!profile) {
      return new Response(
        JSON.stringify({ error: 'Invalid token: No owner found with this token' }),
        { status: 401, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }

    // Create service client for database operations
    const supabaseService = createClient(supabaseUrl, supabaseServiceKey, {
      auth: {
        autoRefreshToken: false,
        persistSession: false
      },
      global: {
        headers: {
          'Authorization': `Bearer ${supabaseServiceKey}`
        }
      }
    })

    const { method } = req
    const body = await req.json()

    if (method === 'POST') {
      // Create new pitch
      const { complex_id, ...pitchData } = body

      // Verify ownership of complex
      const { data: complex } = await supabaseService
        .from('complexes')
        .select('owner_id')
        .eq('id', complex_id)
        .single()

      if (!complex || complex.owner_id !== profile.id) {
        return new Response(
          JSON.stringify({ error: 'Unauthorized: Not the owner of this complex' }),
          { status: 403, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }

      const { data, error } = await supabaseService
        .from('pitches')
        .insert([{ ...pitchData, complex_id }])
        .select()
        .single()

      if (error) {
        return new Response(
          JSON.stringify({ error: error.message }),
          { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }

      return new Response(
        JSON.stringify({ data }),
        { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    } else if (method === 'PUT') {
      // Update existing pitch
      const { id, ...updateData } = body

      // Verify ownership through complex
      const { data: pitch } = await supabaseService
        .from('pitches')
        .select('complex_id')
        .eq('id', id)
        .single()

      if (!pitch) {
        return new Response(
          JSON.stringify({ error: 'Pitch not found' }),
          { status: 404, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }

      // Check complex ownership
      const { data: complex } = await supabaseService
        .from('complexes')
        .select('owner_id')
        .eq('id', pitch.complex_id)
        .single()

      if (!complex || complex.owner_id !== profile.id) {
        return new Response(
          JSON.stringify({ error: 'Unauthorized: Not the owner of this pitch' }),
          { status: 403, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }

      const { data, error } = await supabaseService
        .from('pitches')
        .update(updateData)
        .eq('id', id)
        .select()
        .single()

      if (error) {
        return new Response(
          JSON.stringify({ error: error.message }),
          { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
        )
      }

      return new Response(
        JSON.stringify({ data }),
        { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      )
    }

    return new Response(
      JSON.stringify({ error: 'Method not allowed' }),
      { status: 405, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    )
  } catch (error) {
    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    )
  }
})
